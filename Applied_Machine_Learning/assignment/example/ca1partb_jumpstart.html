from sklearn.model_selection import GridSearchCV
from sklearn.neighbors import KNeighborsRegressor
from sklearn.model_selection import cross_val_score
from sklearn.metrics import r2_score

params = {
    'n_neighbors': [3, 5, 10, 20, 100],
}

knnmodel = KNeighborsRegressor()
grid_search = GridSearchCV(knnmodel, params, cv=5)
grid_search.fit(X_train, y_train)
best_params = grid_search.best_params_
print("Best parameters:", best_params)
best_model = grid_search.best_estimator_

cv_score = cross_val_score(best_model, X_train, y_train, cv=5)
#print(cv_score)
ytest_hat = best_model.predict(X_test)

print("Train Score=",best_model.score(X_train,y_train))
print("Cross-validation score=",cv_score.mean())
print("Test score=",r2_score(y_test,ytest_hat))
